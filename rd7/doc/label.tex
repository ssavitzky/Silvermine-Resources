% Coordinate calculations.  Isn't arithmetic in TeX fun??  No.
%
% We want 4x1.5, which in points is 288 * 108
%
%	use offset to start the outer box at 0,0
%	so inner box is at LS, LS
%
\newcommand{\TW}{288}		% Total Width
\newcommand{\TH}{104}		% Total Height
\newcommand{\EB}{8}		% Empty Border
\newcommand{\LS}{2}		% Line Spacing
\newcommand{\TBH}{15}		% Top Box Height
  \newcounter{bw}\setcounter{bw}{\TW}
  \addtocounter{bw}{-\EB}
  \addtocounter{bw}{-\EB}
\newcommand{\BW}{\value{bw}}	% Box Width		= TW - 2*EB
  \newcounter{bh}\setcounter{bh}{\TH}
  \addtocounter{bh}{-\EB}
  \addtocounter{bh}{-\EB}
\newcommand{\BH}{\value{bh}}	% Box Height		= TH - 2*EB
  \newcounter{ibw}\setcounter{ibw}{\BW}
  \addtocounter{ibw}{-\LS}
  \addtocounter{ibw}{-\LS}
\newcommand{\IBW}{\value{ibw}}	% Inner Box Width	= BW - 2*LS
\newcommand{\IBX}{\LS}		% Inner Box X Position	= LS
  \newcounter{tby}\setcounter{tby}{\BH}
  \addtocounter{tby}{-\LS}
  \addtocounter{tby}{-\TBH}
\newcommand{\TBY}{\value{tby}}	% Top Box Y Position	= BH - LS - TBH
  \newcounter{bbh}\setcounter{bbh}{\TBY}
  \addtocounter{bbh}{-\LS}
  \addtocounter{bbh}{-\LS}
\newcommand{\BBH}{\value{bbh}}	% Bottom Box Height	= TBY - 2*LS
\newcommand{\BBY}{\LS}		% Bottom Box Y Position = LS
%
